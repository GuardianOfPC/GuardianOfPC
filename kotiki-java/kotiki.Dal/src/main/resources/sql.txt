create table owners
(
    id            serial
        constraint owners_pk
            primary key,
    date_of_birth date,
    name          varchar
);

alter table owners
    owner to postgres;

create table cats
(
    id            serial
        constraint cats_pk
            primary key,
    name          varchar,
    date_of_birth date,
    breed         varchar,
    color         varchar,
    owner_id      integer
        constraint cats_owners_id_fk
            references owners
            on update cascade on delete cascade,
    friends_id    integer
        constraint cats_friends
            references cats
);

alter table cats
    owner to postgres;
